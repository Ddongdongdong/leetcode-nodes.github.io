class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        dummy1=ListNode(0)
        dummy2=ListNode(-1)
        p1,p2=dummy1,dummy2
        p=head
        while p:
            if p.val<x:
                p1.next=p
                p1=p1.next
            else:
                p2.next=p
                p2=p2.next
        
            tt=p.next
            p.next=None
            p=tt
        p1.next=dummy2.next

将这个链表给的值分为两个链表再合并，主要就是通过这种方式将链表分解。
